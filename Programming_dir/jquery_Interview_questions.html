Jquery实现在页面上拖动元素 

 $(document).ready(function ()
 {
 var mousex = 0, mousey = 0;
 var divLeft, divTop;
 $('.dragMe').mousedown(function(e)
 {//鼠标按下时，获取相应的坐标；
 var offset = $(this).offset();
 divLeft = parseInt(offset.left,10);
 divTop = parseInt(offset.top,10);
 mousey = e.pageY;
 mousex = e.pageX;
 $(this).bind('mousemove',dragElement);
 });//鼠标移动时，绑定dragElement函数；
 
function dragElement(event)
 {
 var left = divLeft + (event.pageX - mousex);
 var top = divTop + (event.pageY - mousey);
 $(this).css(
 {
 'top' : top + 'px',
 'left' : left + 'px',
 'position' : 'absolute'//必须将位置设置成abolute,不然无法移动（这一点有谁知道为何么？）
 });
 return false;
 }
 $(document).mouseup(function()
 {
 $('.dragMe').unbind('mousemove');
 });
 
});

     cur.after('<span>Mandatory field!</span>');//则在输入框后添加错题提示信息；
     dataValid=false;
    }else{dataValid=true;}
  });
  $('#check').click(function(){//点击了Check按钮之后，执行....
  if(dataValid)
     {
      $('#info').html('Validation OK');//提示通过验证
     }});
 });

Jquery创建简易的选项卡导航界面 

$(document).ready(function ()
   {//当页面载入时，自动运行以下代码；
    $('.tabContent:gt(0)').hide();//将所有的选项卡内容隐藏，gt(x)表示改元素中第x后的所有元素，0表示第一个；
    $('.tabHeader > li:eq(0)').addClass('active');//默认给第一个选项卡添加深色；
    $('.tabHeader > li').click(showHideTabs);//点击某个选项卡之后，运行showHideTabs函数；
   });
   function showHideTabs()
   {
    var allLi = $('.tabHeader > li').removeClass('active')//将所有选项卡变为浅色，并让allLi表示为所有的选项卡；
    $(this).addClass('active');//给点击中的选项卡添加深色；
     
    var index = allLi.index(this);//找出点击中的选项卡处于队列中的第几个；
    $('.tabContent:visible').hide();//让原先的选项卡内容隐藏；
    $('.tabContent:eq('+index+')').show();//显示被点中的选项卡内容；
   }

用Jquery创建简单的浮动菜单 

$(document).ready(function ()
  {
   $('.menuItem').hide();//先让菜单默认为隐藏的；
   $(window).scroll(floatDiv);//浏览器滚动时，运行floatDiv函数
   floatDiv();//页面加载成功时，也运行floatDivh函数，哈哈，这个函数出镜率很高吧
  });
  function floatDiv()//函数的作用大概是定位浮动框的位置；
  {
   $('#floatingBox').animate({top: $(document).scrollTop() + "px"},{duration:250,queue:false});
  }
  $('span.menu').click(function()//如果点击了
  {
   $(this).next('ul').slideToggle('fast');//就迅速改变菜单的显示状态（在hide和show中切换）
  });
